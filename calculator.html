🧮 Calculator

A terminal-based calculator with basic operations, power, square root, and history tracking.

What I Learned

The trickiest part was keeping the program running until the user wants to quit. I used a while True loop for that. I also learned how to organize code into functions so each operation is clean and reusable.

while True loop if / elif float() functions math.sqrt() history list
calculator.py 
import math
import os

def add(a, b):
    return a + b

def subtract(a, b):
    return a - b

def multiply(a, b):
    return a * b

def divide(a, b):
    if b == 0:
        return "Cannot divide by zero"
    return a / b

def power(a, b):
    return a ** b

def square_root(a):
    if a < 0:
        return "Cannot take square root of a negative number"
    return math.sqrt(a)

def clear_screen():
    os.system("cls" if os.name == "nt" else "clear")

def show_menu():
    print("\n--- Calculator ---")
    print("1. Add")
    print("2. Subtract")
    print("3. Multiply")
    print("4. Divide")
    print("5. Power (a ** b)")
    print("6. Square Root (√a)")
    print("7. History")
    print("8. Clear Screen")
    print("9. Exit")

history = []

while True:
    show_menu()
    choice = input("Choose an option (1-9): ").strip()

    if choice == "9":
        print("Goodbye!")
        break

    if choice == "8":
        clear_screen()
        continue

    if choice == "7":
        if len(history) == 0:
            print("History is empty.")
        else:
            print("\n--- History ---")
            for item in history:
                print(item)
        continue

    if choice == "6":
        a = float(input("Enter a number: "))
        result = square_root(a)
        print("Result:", result)
        if isinstance(result, (int, float)):
            history.append(f"sqrt({a}) = {result}")
        else:
            history.append(f"sqrt({a}) -> {result}")
        continue

    a = float(input("Enter first number: "))
    b = float(input("Enter second number: "))

    if choice == "1":
        result = add(a, b)
        expr = f"{a} + {b} = {result}"
    elif choice == "2":
        result = subtract(a, b)
        expr = f"{a} - {b} = {result}"
    elif choice == "3":
        result = multiply(a, b)
        expr = f"{a} * {b} = {result}"
    elif choice == "4":
        result = divide(a, b)
        expr = f"{a} / {b} = {result}"
    elif choice == "5":
        result = power(a, b)
        expr = f"{a} ** {b} = {result}"
    else:
        print("Invalid choice")
        continue

    print("Result:", result)
    history.append(expr)
